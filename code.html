    function FizzBuzzA(value1, value2){
        let returnValue = "";
        for(let loop = 1; loop <=100; loop++){
            if (loop%value1 == 0 && loop%value2 == 0) {
                returnValue += 'FizzBuzz  ';
            } 
            else if (loop%value1 == 0){
                returnValue +='Fizz  ';
            }
            else if (loop%value2 == 0){
                returnValue =+ 'Buzz  ';
            }
            else{
                returnValue += loop + ' ';
            }
        }
        return returnValue;
    }
